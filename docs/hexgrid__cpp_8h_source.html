128 p (idx_t idir, idx_t inode, idx_t colidx, idx_t rowidx, double hx, double hy) {

168 shg (double x, double y, idx_t idir, idx_t inode,
169 idx_t c, idx_t r, double hx, double hy) {
170 switch (inode) {
171 case 3 :
172 if (idir == 0) {
173 return ((1. / hx) * ((y - p(1,0,c,r,hx,hy)) / hy));
174 }
175 else if (idir == 1) {
176 return (((x - p(0,0,c,r,hx,hy)) / hx) * (1. / hy));
177 }
178 break;
179 case 2 :
180 if (idir == 0) {
181 return ((1. / hx) * ((1. - (y - p(1,0,c,r,hx,hy)) / hy)));
182 }
183 else if (idir == 1) {
184 return (((x - p(0,0,c,r,hx,hy)) / hx) * (- 1. / hy));
185 }
186 break;
187 case 1 :
188 if (idir == 0) {
189 return ((- 1. / hx) * ((y - p(1,0,c,r,hx,hy)) / hy));
190 }
191 else if (idir == 1) {
192 return ((1. - (x - p(0,0,c,r,hx,hy)) / hx) * (1. / hy));
193 }
194 break;
195 case 0 :
196 if (idir == 0) {
197 return ((- 1. / hx) * (1. - (y - p(1,0,c,r,hx,hy)) / hy));
198 }
199 else if (idir == 1) {
200 return ((1. - (x - p(0,0,c,r,hx,hy))/ hx) * (- 1. / hy));
201 }
202 break;
203 }
