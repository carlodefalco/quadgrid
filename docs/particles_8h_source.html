175 remove_in_region (std::function<bool (double, double)> fun) {
176 std::vector<idx_t> vin{};
177 for (idx_t i = 0; i < num_particles; ++i) {
178 if (fun (x[i], y[i])) {
179 vin.push_back (i);
180 }
181 }
182
183 for (auto &dprop : dprops) {
184 for (auto id = vin.rbegin (); id != vin.rend (); ++id) {
185 dprop.second.erase (dprop.second.begin () + (*id));
186 }
187 }
188
189 for (auto &iprop : iprops) {
190 for (auto id = vin.rbegin (); id != vin.rend (); ++id) {
191 iprop.second.erase (iprop.second.begin () + (*id));
192 }
193 }
194
195 for (auto id = vin.rbegin (); id != vin.rend (); ++id) {
196 x.erase (x.begin () + (*id));
197 y.erase (y.begin () + (*id));
198 --num_particles;
199 }
200 };
